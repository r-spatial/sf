# functions only exported to be used internally by stars

functions only exported to be used internally by stars

## Usage

``` r
.get_layout(bb, n, total_size, key.pos, key.width, mfrow = NULL, main = NULL)

.degAxis(side, at, labels, ..., lon, lat, ndiscr, reset)

.image_scale(
  z,
  col,
  breaks = NULL,
  key.pos,
  add.axis = TRUE,
  at = NULL,
  ...,
  axes = FALSE,
  key.length,
  logz = FALSE,
  lab = "",
  cex.axis = par("cex.axis")
)

.image_scale_factor(
  z,
  col,
  key.pos,
  add.axis = TRUE,
  ...,
  axes = FALSE,
  key.width,
  key.length,
  cex.axis = par("cex.axis")
)
```
